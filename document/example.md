## ตัวอย่างข้อมูลเพิ่มเติม

### ตัวอย่างการใช้งาน Query พื้นฐาน

```sql
-- ดูรายการสั่งซื้อพร้อมรายละเอียดลูกค้า
SELECT o.order_number, u.first_name, u.last_name, o.total_amount, o.order_status
FROM orders o
JOIN users u ON o.user_id = u.id
WHERE o.created_at >= '2024-02-01'
ORDER BY o.created_at DESC;

-- ดูค่าส่วนกลางที่ยังไม่ได้ชำระ
SELECT u.unit_number, ur.first_name, ur.last_name, mf.total_amount, mf.due_date
FROM monthly_fees mf
JOIN units u ON mf.unit_id = u.id
JOIN residents r ON u.id = r.unit_id AND r.is_primary = TRUE
JOIN users ur ON r.user_id = ur.id
WHERE mf.status = 'unpaid' AND mf.due_date < CURDATE()
ORDER BY mf.due_date ASC;

-- ดูคำขอบริการที่รอดำเนินการ
SELECT sr.request_number, u.first_name, u.last_name, sr.title, sr.priority, sr.created_at
FROM service_requests sr
JOIN users u ON sr.user_id = u.id
WHERE sr.status IN ('pending', 'assigned')
ORDER BY sr.priority DESC, sr.created_at ASC;
```

### สถิติและรายงาน

```sql
-- สรุปยอดขายรายเดือน
SELECT 
    DATE_FORMAT(created_at, '%Y-%m') as month,
    COUNT(*) as total_orders,
    SUM(final_amount) as total_revenue
FROM orders 
WHERE order_status = 'delivered'
GROUP BY DATE_FORMAT(created_at, '%Y-%m')
ORDER BY month DESC;

-- อัตราการอ่านประกาศ
SELECT 
    a.title,
    COUNT(DISTINCT u.id) as total_residents,
    COUNT(ar.user_id) as readers_count,
    ROUND((COUNT(ar.user_id) / COUNT(DISTINCT u.id)) * 100, 2) as read_percentage
FROM announcements a
CROSS JOIN users u
LEFT JOIN announcement_reads ar ON a.id = ar.announcement_id AND u.id = ar.user_id
WHERE u.role = 'resident' AND a.status = 'published'
GROUP BY a.id, a.title
ORDER BY read_percentage DESC;
```
